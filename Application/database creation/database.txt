CREATE DATABASE bank;

CREATE TABLE tellers(Teller_ID INT(10) NOT NULL AUTO_INCREMENT , Username VARCHAR(20) NOT NULL, Password VARCHAR(20) NOT NULL,
Fname VARCHAR(20), Mname VARCHAR(20), Lname VARCHAR(30), DOB DATE, Address VARCHAR(100),
Pnumber INT(10), Email VARCHAR(30), city VARCHAR(20), branch VARCHAR(20), PRIMARY KEY (Teller_ID)) ENGINE = InnoDB;

CREATE TABLE Managers(Manager_ID INT(10) NOT NULL AUTO_INCREMENT , Username VARCHAR(20) NOT NULL, Password VARCHAR(20) NOT NULL,
Fname VARCHAR(20), Mname VARCHAR(20), Lname VARCHAR(30), DOB DATE, Address VARCHAR(100),
Pnumber INT(10), Email VARCHAR(30), city VARCHAR(20), branch VARCHAR(20), PRIMARY KEY (Manager_ID)) ENGINE = InnoDB;

CREATE TABLE Customers(Customer_ID INT(10) NOT NULL AUTO_INCREMENT ,Fname VARCHAR(20), Mname VARCHAR(20), Lname VARCHAR(30), Age INT(10),
NIC VARCHAR(15), Email VARCHAR(30), Pnumber INT(10), Address VARCHAR(100), city VARCHAR(20), monthlyIncome INT(15), PRIMARY KEY (Customer_ID)) ENGINE = InnoDB;

CREATE TABLE accounts(Account_ID INT(10) NOT NULL AUTO_INCREMENT , Account_Type VARCHAR(20) NOT NULL , Customer_ID INT(10) NOT NULL ,
Balance INT NOT NULL , createdAt DATE, PRIMARY KEY (Account_ID),  FOREIGN KEY (Customer_ID) REFERENCES Customers(Customer_ID)) ENGINE = InnoDB;
